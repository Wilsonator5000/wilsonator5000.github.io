//* Checks if the passed string matches a string version

//* of the passed number.

//* @returns: `true` if they match, `false` if they

//*     don't.

//*

//* @example: ```

//* does('3', match 3) // true

//* does('56', match 3) // false

//* does('qwerty', match 3) // false

//* ```

//*

//* @author: Wilson Gramer

//* @see-also: my-inline-function

function does(_ string str, (match) number x) returns bool: {

    return (try? string(x)) is str

}

 

const is-match = does('42', match 7*6)

print(is-match)

from colors import color

 

class Person {

 

    string name

    int age

    color hair-color

 

    init(string name, int age, (hair) color hair-color): {

        self.name = name

        self.age = age

        self.hair-color = hair-color

    }

 

    bool has-black-hair: self.hair-color is .black

 

    function greeting() returns string: {

        if self.age > 12 {

            return 'Hi there, {self.name}!'        

        } else if it.is(between 3..<12) {

            return 'Hey, kiddo!'

        } else {

            return 'Hello, little guy!'

       }

    }

}

 

let person = Person(name = 'Bob', age = 25, hair #654321)

print(person.greeting())
